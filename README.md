# pop-crosshair

Add a popup crosshair in your game

CONFIG (client.lua):

  local sprite = {
      dictionary = "helicopterhud",
      texture = "hud_dest",
      rotate = 43.0,
      firstWidth = 0.02,
      firstHeight = 0.02,
      step = 0.02,
      lastWidth = 0.08,
      lastHeight = 0.08,
      rgba = {255, 255, 255, 255},
      stepInterval = 1
  }
